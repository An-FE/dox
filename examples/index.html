param: String  nothing a string that does nothing
return: Object  exports for chaining
api: public
 
var output = exports.output = function(nothing){
    for (var i = 0, len = files.length; i < len; ++i) {
        var file = files[i];
        if (file.match(/\.js$/)) {
            fs.readFile(file, function(err, str){
               if (err) {
                   throw err;
               } else {
                   sys.puts(str); 
               }
            });
        }
    }
    return this;
}
